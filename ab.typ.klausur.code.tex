\newcommand{\ab@kl@dauer}{90}
\newcommand{\ab@kl@hmft}{0}

\pgfkeys{
	/absetup/.cd,
	dauer/.code=\ab@kl@renewTimes{#1}{\ab@kl@hmft},
	hmft/.code=\ab@kl@renewTimes{\ab@kl@dauer}{#1},
}

\newcommand{\ab@kl@renewTimes}[2]{%
	\renewcommand{\ab@kl@dauer}{#1}\renewcommand{\ab@kl@hmft}{#2}%
	\pfgmathtruncatemacro{\ab@kl@hmt}{\ab@kl@dauer-\ab@kl@hmft}%
}
